\begin{minipage}{\linewidth}

        \item version
        \item author
        \item source location
        \item last package update
        \item depencencies
        \item description
        \item installed version
        \item available version (from abt repository)
      \end{enumerate}
  \end{enumerate}
  \vspace{0.05em}
