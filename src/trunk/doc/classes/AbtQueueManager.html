actionPackageQueue( package, queue, action="add" )

Add/Remove a given package to the given queue. If adding a package already in the queue then it will not be added twice and return succes.

PARAM String - the package to be added/removed. PARAM String - the queue.

RETURN boolean - true if action succeeds, otherwise false.

 63:     end

 66:       if ( log = File.new( queueFile, File::WRONLY|File::APPEND|File::CREAT, 0644 ) )
 67:         # pickup queue contents to ensure no duplicates.
 68:         checkingQueue = IO.readlines( queueFile )
 69:     
 70:         # check if package exists, otherwise add.
 71:         if ( ! checkingQueue.collect{ |i| i.split( '|' )[0] }.include?( package ) )
 72:           log.puts "#{package}|#{$TIMESTAMP}"
 73:           logger.logToJournal( "Added #{package} to #{queue} queue." )
 74:         else
 75:           logger.logToJournal( "Did not add #{package} to #{queue}, already exists." )
 76:         end
 77:    
 78:         log.close
 79:         return true
 80:       end
 81:     end
 82: 
 83:     if ( action == "remove" )
 84:       # remove entry from given queue.
 85:       if ( log = File.new( queueFile, File::WRONLY|File::APPEND|File::CREAT, 0644 ) )
 86:         # use temp file to filter out entry to be removed.
 87:         temp = File.new(queueFile + ".tmp", "a+")
 88:         
 89:         # now check for line to be removed.
 90:         IO.foreach( queueFile ) do |line|
 91:           entryName = line.split( '|' )[0]
 92:           if ( entryName != package.downcase )
 93:             temp.puts line
 94:           end
 95:         end
 96:         
 97:         temp.close
 98:         FileUtils.mv( temp.path, queueFile )
 99:       end
100:

103:     end
