     # File libs/abtlogmanager.rb, line 186
186:   def log_package_install(package)
187:                 renames    = Hash.new  # to hold installwatch identified renames.
188:                 duplicates = Hash.new  # to hold entries for duplicate checking.
189: 
190:     # some dirs we will not add to an install log.
191:     excluded_pattern = Regexp.new("^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys|#{$DEFAULT_PREFIX}/usr/src)+")
192:     
193:     # our log locations.
194:     installLog = get_log(package, 'install')
195:     tmpInstallLog = get_log(package, 'tmpinstall')
196:     
197:     # get the installed files from the tmp file
198:     # into our install log.
199:     if (File.exist?(tmpInstallLog))
200:       installFile = open(installLog, 'w')
201:       
202:       # include only the file names from open calls
203:       # and not part of the excluded range of directories.
204:       IO.foreach(tmpInstallLog) do |line|
205:                                 if (line.split[1] == 'rename')
206:                                         # they renamed the line, save this entry for cleaning
207:                                         # after install log is closed. Hash key is the original
208:                                         # installed file name, Hash value is the new file name.
209:                                         if !(renames.has_key?(line.split[2]))
210:                                                 renames[line.split[2]] = line.split[3] 
211:                                         end
212:         elsif (line.split[1] == 'open') and File.exist?(line.split[2])
213:           if !(line.split[2] =~ excluded_pattern)
214:                                                 # possible install log addition.
215:                                                 if !duplicates.key?(line.split[2])
216:                                                         # add to duplicate tracking hash and install log.
217:                                                         duplicates[line.split[2]] = line.split[2]
218:                                                         installFile << "#{line.split[2]}\n"
219:                                                 end
220:           end
221:         end 
222:       end
223:                         
224:                         check_for_file_renames(renames, installLog) if !renames.empty?
225:     else
226:       # no tmp install file, thus no install running.
227:       return false
228:     end
229:     
230:     return true;

144:   def log_package_integrity(package)
145:                 logger = Logger.new($JOURNAL)
146:   
147:     # our log locations.
148:     installLog = get_log(package, 'install')
149:     integrityLog = get_log(package, 'integrity')
150:     
151:     # get the installed files from the tmp file
152:     # into our install log.
153:     if (File.exist?(installLog))
154:       installFile   = open(installLog, 'r')
155:       integrityFile = open(integrityLog, 'w')
156: 
157:       # get the integrity for each file, initially just permissions.      
158:       IO.foreach(installLog) do |line|
159:                                 if File.exist?(line.chomp)
160:                                         status = File.stat(line.chomp)
161:                                         octal  = sprintf("%o", status.mode)
162:                                         integrityFile << "#{line.chomp}:#{octal}\n"
163:                                 else
164:                                         logger.info("[log_package_integrity] - #{line} is not installed, no worries, I will take care of everything...")
165:                                 end 
166:       end
167:       
168:       installFile.close
169:       integrityFile.close
170:     else
171:       return false  # no install log!
172:     end
173:     
174:     return true;
