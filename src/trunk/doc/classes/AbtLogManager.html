     # File abtlogmanager.rb, line 141
141:   def log_package_install( package )
142:     # some dirs we will not add to an install log.
143:     excluded_pattern = Regexp.new( "^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys)+" )
144:     badLine = false  # used to mark excluded lines from installwatch log.
145:     
146:     # our log locations.
147:     installLog = get_log( package, 'install' )
148:     tmpInstallLog = get_log( package, 'tmpinstall' )
149:     
150:     # get the installed files from the tmp file
151:     # into our install log.
152:     if ( File.exist?( tmpInstallLog ) )
153:       installFile = open( installLog, 'w')
154:       
155:       # include only the file names from open calls
156:       # and not part of the excluded range of directories.
157:       IO.foreach( tmpInstallLog ) do |line|
158:         if ( line.split[1] == 'open' )
159:           if ( line.split[2] =~ excluded_pattern )
160:             badLine = true
161:           else
162:             badLine = false
163:           end
164:           
165:           if ( !badLine )
166:             installFile << "#{line.split[2]}\n"
167:           end
168:         end 
169:       end
170:       
171:       installFile.close
172:     end
173:     
174:     return true;
175:   end
