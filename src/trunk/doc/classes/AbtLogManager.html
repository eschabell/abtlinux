     # File AbtLogManager.rb, line 106
106:   def logPackageInstall( package )
107:     # some dirs we will not add to an install log.
108:     excluded_pattern = 
109:     Regexp.new( "^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys)+" )
110:     
111:     require package
112:     sw = eval( "#{package.capitalize}.new" )
113:     details = sw.details
114:     badLine = false  # used to mark excluded lines from installwatch log.
115:     
116:     # our log locations.
117:     installLog = "#{$PACKAGE_INSTALLED}/#{details['Source location']}" + 
118:       "/#{details['Source location']}.install"
119:     tmpInstallLog = "#{$ABT_TMP}/#{details['Source location']}.watch"
120:     
121:     # get the installed files from the tmp file
122:     # into our install log.
123:     if ( File.exist?( tmpInstallLog ) )
124:       installFile = open( installLog, 'w')
125:       
126:       # include only the file names from open calls
127:       # and not part of the excluded range of directories.
128:       IO.foreach( tmpInstallLog ) do |line|
129:         if ( line.split[1] == 'open' )
130:           if ( line.split[2] =~ excluded_pattern )
131:             #self.logToJournal( "DEBUG: Found bad logLine!" )
132:             badLine = true
133:           else
134:             badLine = false
135:           end
136:           
137:           if ( !badLine )
138:             #self.logToJournal( "DEBUG: adding line to installFile!")
139:             installFile.puts line.split[2]
140:           end
141:         end 
142:       end
143:       
144:       installFile.close
145:     end
146:     
147:     return true;
148:   end
