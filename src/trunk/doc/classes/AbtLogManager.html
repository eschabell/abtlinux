PARAM String - Package name.

[ show source ]

     # File libs/abtlogmanager.rb, line 139
139:   def log_package_install( package )
140:     # some dirs we will not add to an install log.
141:     excluded_pattern = Regexp.new( "^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys)+" )
142:     badLine = false  # used to mark excluded lines from installwatch log.
143:     
144:     # our log locations.
145:     installLog = get_log( package, 'install' )
146:     tmpInstallLog = get_log( package, 'tmpinstall' )
147:     
148:     # get the installed files from the tmp file
149:     # into our install log.
150:     if ( File.exist?( tmpInstallLog ) )
151:       installFile = open( installLog, 'w')
152:       
153:       # include only the file names from open calls
154:       # and not part of the excluded range of directories.
155:       IO.foreach( tmpInstallLog ) do |line|
156:         if ( line.split[1] == 'open' )
157:           if ( line.split[2] =~ excluded_pattern )
158:             badLine = true
159:           else
160:             badLine = false
161:           end
162:           
163:           if ( !badLine )
164:             installFile << "#{line.split[2]}\n"
165:           end
166:         end 
167:       end
168:       
169:       installFile.close
170:     else
171:       # no tmp install file, thus no install running.
172:       return false
173:     end
174:     
175:     return true;
176:   end
