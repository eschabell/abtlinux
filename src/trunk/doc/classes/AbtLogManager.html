 79:     # some dirs we will not add to an install log.
 80:     excluded_pattern = Regexp.new( "^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys)+" )
 81:     
 82:     require package
 83:     sw = eval( "#{package.capitalize}.new" )
 84:     details = sw.details
 85:     badLine = false  # used to mark excluded lines from installwatch log.
 86:     
 87:     # our log locations.
 88:     installLog = "#{$PACKAGE_INSTALLED}/#{details['Source location']}/#{details['Source location']}.install"
 89:     tmpInstallLog = "#{$ABT_TMP}/#{details['Source location']}.watch"
 90:     
 91:     # get the installed files from the tmp file
 92:     # into our install log.
 93:     if ( File.exist?( tmpInstallLog ) )
 94:       installFile = open( installLog, 'w')
 95:       
 96:       # include only the file names from open calls
 97:       # and not part of the excluded range of directories.
 98:       IO.foreach( tmpInstallLog ) do |line|
 99:         if ( line.split[1] == 'open' )
100:           #self.logToJournal( "DEBUG: checking: #{line.split[2]} against #{excluded_pattern}." )
101:           if ( line.split[2] =~ excluded_pattern )
102:             #self.logToJournal( "DEBUG: Found bad logLine!" )
103:             badLine = true
104:           else
105:             badLine = false
106:             #self.logToJournal( "DEBUG: #{excluded_pattern} not matching #{line.split[2]}")
107:           end
108:           
109:           if ( !badLine )
110:             #self.logToJournal( "DEBUG: adding line to installFile!")
111:             installFile.puts line.split[2]
112:           end
113:         end 
114:       end
115:       
116:       installFile.close
117:     end
118:     
119:     # cleanup the tmp files.
