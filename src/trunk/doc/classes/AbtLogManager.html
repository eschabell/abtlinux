PARAM String - Package name.

RETURN boolean - True if install log created successfully, otherwise false.

[ show source ]

     # File libs/abtlogmanager.rb, line 144
144:   def log_package_install(package)
145:     # some dirs we will not add to an install log.
146:     excluded_pattern = Regexp.new("^(/dev|/proc|/tmp|/var/tmp|/usr/src|/sys)+")
147:     badLine = false  # used to mark excluded lines from installwatch log.
148:     
149:     # our log locations.
150:     installLog = get_log(package, 'install')
151:     tmpInstallLog = get_log(package, 'tmpinstall')
152:     
153:     # get the installed files from the tmp file
154:     # into our install log.
155:     if (File.exist?(tmpInstallLog))
156:       installFile = open(installLog, 'w')
157:       
158:       # include only the file names from open calls
159:       # and not part of the excluded range of directories.
160:       IO.foreach(tmpInstallLog) do |line|
161:         if (line.split[1] == 'open')
162:           if (line.split[2] =~ excluded_pattern)
163:             badLine = true
164:           else
165:             badLine = false
166:           end
167:           
168:           if (!badLine)
169:             installFile << "#{line.split[2]}\n"
170:           end
171:         end 
172:       end
173:       
174:       installFile.close
175:     else
176:       # no tmp install file, thus no install running.
177:       return false
178:     end
179:     
180:     return true;
181:   end
log_package_integrity(package)

Provides logging of the integrity of all installed files for the given package. Will be called as part of the logging done during the install phase.

PARAM String - Package name.

RETURN boolean - True if integrity log created successfully, otherwise false.

[ show source ]

     # File libs/abtlogmanager.rb, line 107
107:   def log_package_integrity(package)
108:     
109:     # our log locations.
110:     installLog = get_log(package, 'install')
111:     integrityLog = get_log(package, 'integrity')
112:     
113:     # get the installed files from the tmp file
114:     # into our install log.
115:     if (File.exist?(installLog))
116:       installFile   = open(installLog, 'r')
117:       integrityFile = open(integrityLog, 'w')
118: 
119:       # get the integrity for each file, initially just permissions.      
120:       IO.foreach(installLog) do |line|
121:         status = File.stat(line.chomp)
122:         octal  = sprintf("%o", status.mode)
123:         integrityFile << "#{line.chomp}:#{octal}\n"
124:       end
125:       
126:       installFile.close
127:       integrityFile.close
128:     else
129:       return false  # no install log!
130:     end
131:     
132:     return true;
133:   end
